52def ProfileGenerator(length,Profile):
53 """! Profile generator function, returns a dataframe with the profile.
54
55 @param length the length of the dataset to be generated
56 @param Profile A dictionary with name:array pairs where the array has format [value,upto,value,upto,etc....]
57 @param verbose verbose boolean generates more output for debugging
58
59 @return A Pandas dataFrame of length with the specified profile
60 """
61 df = pd.DataFrame(index=range(length)) # instantiate dataframe
62 for i,x in enumerate(Profile): # for all values that need to be put in column, in this case Ta, Pe, Ti
63 for j,y in enumerate(Profile[x]): # for all profile points in Ta, Pe, Ti
64 if(j==0): #if it is the first element, fill from element 0 up to the next change
65 df.loc[0:Profile[x][j+1],(x)] = y # create zero column to contain prediction
66 if(j%2==0): # if modulus 2 of index is true; aka even numbered indexes
67 df.loc[Profile[x][j-1]:Profile[x][j+1],(x)] = y # fill from last change up to this change with fixed value y
68 return df

298 """! Do parameter optimization with scipy.optimize
299
300 @param constraints dictionary of arrays with [min,max] values to constraint the parameter search to
301 example: SMap = {"U":[0.01,45.0],"C":[100,200]}. If not given, all in solveList will be optimized without constraints

303 @param steps amount of dataset steps to calculate in the error function
304 @param iterations amount of allowed solver iterations, default set to 1000
305 @param solver solving algorithm to be used, default nelder-mead
306 @param lsq treat problem as least-squares problem

308 @return An array with the best parameters found

310 if err is not None:
311 x0 = []

357 def parameterSearchMap(self,constraints=None, err=None, ims = 4, steps = None):# format of "coef":[min,max], etc
358 """! Do parameter search over a defined area, results in steps*(constants^ims) iterations
359
360 @param constraints dictionary of arrays with [min,max] values to constraint the parameter search to
361 example: SMap = {"U":[0.01,45.0],"C":[100,200]}
362 @param err Name of variable to track as error, make sure this is in the predictors and in the loaded data
363 @param ims Intermediate steps to generate, if 0 and 1 are given, a parameter search for 0, 0.25, 0.5 and 1 will be executed
364
365 @return An array with errors in the format of [index,error,c0,c1,cn....]
366 """
367 if constraints is not None and err is not None:
368 self.constraints = constraints
369 constraintArray = []
370 if self.Verbose >= 1:
371 print("Constraints {} were provided".format(constraints))
372 for i in constraints:
373 Min = constraints[i][0]
374 Max = constraints[i][1]
375 Delta = Max-Min
376 for j in range(ims-2):
377 constraints[i].append(Min+((Delta/ims)*(j+1)))
378 constraints[i] = sorted(constraints[i])
379 constraintArray.append(constraints[i])
380 if self.Verbose >= 1:
381 print("{} will be tested for {}".format(constraints[i],i))
382 pSearchMap = []
383 for i in itertools.product(*constraintArray):
384 pSearchMap.append(i)
385 if self.Verbose >= 1:
386 print("{} parameter sets were generated and will be tested".format(len(pSearchMap)))
387 self.pSearchMap = pSearchMap
388 for i in self.pSearchMap:
389 for j,x in enumerate(self.constraints):
390 self.constants[x] = i[j]
391 if steps is None:
392 steps = self.dataLength
393 self.Simulate(steps,err=err)
394 return self.error
395 else:
396 print("Provide constraints!")

401 def Fiddler(self,curr,previousR,currentR,constantIndex): # wqe want to compare the previous error and previous previous error, to at least determine the best "direction"
402 """! Parameter transformer/gradient descent algorithm
403
404 @param curr current parameter that will be transformed, change is proportional to this.
405 @param previousR previous error array, this is used to determine the best direction to change towards
406 @param currentR current error array, as mentioned, to implement some form of gradient descent
407 @param constantIndex index of the current constant that is modified, needed to know what parameter in error array to look at
408
409 @return A new constant to try
410 """
411 sign = random.randint(0,10)
412 if self.Verbose >= 5:
413 print("###########-- Fiddler input: i:{}, current:{} - {} previous:{}".format(constantIndex,curr,currentR,previousR))
414 if previousR[1] < currentR[1]: #
415 if self.Verbose >= 5:
416 print("Previous set of variables better")
417 if previousR[constantIndex] < currentR[constantIndex]:
418 if sign > 9:
419 sign = 1
420 else:
421 sign = -1
422 else:
423 if sign > 9:
424 sign = -1
425 else:
426 sign = 1
427 else:
428
429 if sign > 5:
430 sign = -1
431 else:
432 sign = 1
433 Amount = (random.randint(1,1000)/999)*0.1
434 return curr + (curr*Amount*sign)

643 @param write if set to false, no data is written to SDF. This is used internally to test errors without tainting SDF
644
645 @return An array with errors in the format of [index,error,c0,c1,cn....] if err is defined, otherwise True
646 """

649 self.loadVariables(self.dictSlice(self.index))

654

657 self.stepIndex()

662 self.dictSlice(self.index)[err],
663 self.predictors[err],
664 *self.dictToArray(self.constants)])
665 except OverflowError:
666 self.error.append([self.index,10**9,*self.dictToArray(self.constants)])
667 if err is not None:
668 return self.error
669 else:
670 return True

682 self.Simulate(steps,err)
683 self.Evolver(err,steps)
